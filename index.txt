<script>
document.addEventListener('DOMContentLoaded', () => {
let steps = document.querySelectorAll('.step');
let nextBtns = document.querySelectorAll('.next-btn');
let celebrateBtn = document.getElementById('celebrate-btn');
let progress = document.getElementById('progress-bar-fill');
let current = 0;

function updateBar(){
progress.style.width = (current/(steps.length-1))*100 + "%";
}
updateBar();

nextBtns.forEach(btn=>{
btn.onclick=()=>{
if(current<steps.length-1){
steps[current].classList.add('opacity-0','pointer-events-none');
current++;
steps[current].classList.remove('opacity-0','pointer-events-none');
updateBar();
}
}
});

celebrateBtn.onclick=()=>{
confetti({particleCount:200,spread:180});
document.getElementById("final-message").style.opacity=1;
}
});
